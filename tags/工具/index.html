3、Zsh 插件 mkdir -p ${ZSH_CUSTOM:-~/.zsh/plugins} # 命令高亮 zsh-syntax-highlighting git clone --depth 1 https://github.com/zsh-users/zsh-syntax-highlighting ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-syntax-highlighting echo '# zsh-syntax-highlighting source ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh ' >> ~/.zshrc # 历史命令 zsh-history-substring-search git clone --depth 1 https://github.com/zsh-users/zsh-history-substring-search ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-history-substring-search echo '# zsh-history-substring-search source ~/.zsh/plugins/zsh-history-substring-search/zsh-history-substring-search.zsh bindkey "$terminfo[kcuu1]" history-substring-search-up bindkey "$terminfo[kcud1]" history-substring-search-down export HISTORY_SUBSTRING_SEARCH_ENSURE_UNIQUE=1 export HISTORY_SUBSTRING_SEARCH_HIGHLIGHT_FOUND="fg=green,bold" ' >> ~/.zshrc #
