Win#

参考：https://zhuanlan.zhihu.com/p/678607774

安装前，终端上运行winget settings命令，先设置如下安装路径，

{
	"installBehavior": {
        "scope": "user",
        "installLocation": "D:\\DevTools"
    }
}

依次运行以下命令，安装对应的软件和配置：

# 升级终端
winget install --id Microsoft.Powershell --source winget
# node&npm
winget install -e --id OpenJS.NodeJS
# restart
node --version 
npm --version
# npm install -g npm@latest # 更新npm版本

# Git
winget install Git.Git
# Vscode
winget install Microsoft.VisualStudioCode
# 7-ZIP
winget install 7zip.7zip
# Typora
winget install appmakes.Typora
# Obsidian
winget install Obsidian.Obsidian
# ClashVerge
winget install ClashVergeRev.ClashVergeRev
# IDM
winget install Tonec.InternetDownloadManager
# Chrome 手动安装
# winget install chrome ?

# 自定义Prompt Starship
winget install Starship.Starship
Add-Content -Path $PROFILE -Value @'
# starship
Invoke-Expression (&starship init powershell)
'@
New-Item -ItemType Directory "$env:USERPROFILE\.config" -Force | Out-Null
starship preset pure-preset -o "$env:USERPROFILE\.config\starship.toml"

# 智能移动 zoxide
winget install ajeetdsouza.zoxide
Add-Content -Path $PROFILE -Value @'
# zoxide
Invoke-Expression (& { (zoxide init powershell | Out-String) })
function zi { cd (zoxide query -i | Out-String).Trim() }
'@

# 模糊搜索
winget install junegunn.fzf
Install-Module PSFzf # 管理员权限
Add-Content -Path $PROFILE -Value @'
# fzf & PSFzf
Import-Module PSFzf
Set-PsFzfOption -PSReadlineChordProvider 'Ctrl+t' -PSReadlineChordReverseHistory 'Ctrl+r'
Set-PSReadLineKeyHandler -Key Tab -ScriptBlock { Invoke-FzfTabCompletion }
'@

# 语法高亮bat 
winget install sharkdp.bat

# 高亮eza
winget install eza-community.eza
Add-Content -Path $PROFILE -Value @'
# eza
function Get-DirectoryContent {
    eza --classify --color-scale --icons=always --group-directories-first --time-style iso @args
}
function Get-GitIgnoredDirectoryContent {
    Get-DirectoryContent --git-ignore --git --git-repos @args
}
Set-Alias -Name lsg -Value Get-GitIgnoredDirectoryContent
function Get-DetialedGitIgnoredDirectoryContent {
    Get-GitIgnoredDirectoryContent --long @args
}
Set-Alias -Name llg -Value Get-DetialedGitIgnoredDirectoryContent
'@

Linux#

# 1、安装 Zsh
sudo apt update && sudo apt install zsh -y
# 验证 Zsh
zsh --version  
# 切换到 zsh（相当于重启终端） chsh -s $(which zsh) 
echo '
# zsh
exec zsh
' >> ~/.bashshrc
source ~/.bashrc

# 2、Zsh主题 starship
curl -sS https://starship.rs/install.sh | sh 
# 配置启动主题，写入以下内容到~/.zshrc
echo 'eval "$(starship init zsh)"' >> ~/.zshrc 
# 配置主题 pure
mkdir -p ~/.config && touch ~/.config/starship.toml
starship preset pure-preset -o ~/.config/starship.toml
source ~/.zshrc

# --- pure
mkdir -p "$HOME/.zsh"
git clone https://github.com/sindresorhus/pure.git "$HOME/.zsh/pure"
echo '
# pure
fpath+=($HOME/.zsh/pure)
autoload -U promptinit; promptinit
prompt pure
' >> ~/.zshrc

# 3、Zsh 插件
mkdir -p ${ZSH_CUSTOM:-~/.zsh/plugins}

# 命令高亮 zsh-syntax-highlighting
git clone --depth 1 https://github.com/zsh-users/zsh-syntax-highlighting ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-syntax-highlighting
echo '
# zsh-syntax-highlighting
source ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh
' >> ~/.zshrc

# 自动补全 zsh-autosuggestions
git clone --depth 1 https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-autosuggestions
echo '
# zsh-autosuggestions
source ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-autosuggestions/zsh-autosuggestions.zsh
' >> ~/.zshrc

# 历史命令 zsh-history-substring-search
git clone --depth 1 https://github.com/zsh-users/zsh-history-substring-search ${ZSH_CUSTOM:-~/.zsh/plugins}/zsh-history-substring-search
echo '
# zsh-history-substring-search
source ~/.zsh/plugins/zsh-history-substring-search/zsh-history-substring-search.zsh
bindkey "$terminfo[kcuu1]" history-substring-search-up
bindkey "$terminfo[kcud1]" history-substring-search-down
export HISTORY_SUBSTRING_SEARCH_ENSURE_UNIQUE=1
export HISTORY_SUBSTRING_SEARCH_HIGHLIGHT_FOUND="fg=green,bold"
' >> ~/.zshrc

# 模糊搜索 fzf
git clone --depth 1 https://github.com/junegunn/fzf.git ${ZSH_CUSTOM:-~/.zsh/plugins}/fzf
${ZSH_CUSTOM:-~/.zsh/plugins}/fzf/install
# sudo apt install fzf
# echo "source <(fzf --zsh)" >> ~/.zshrc

# 智能移动 zoxide
curl -sSfL https://ghproxy.net/https://raw.githubusercontent.com/ajeetdsouza/zoxide/main/install.sh | sh
echo -e '
# zoxide
export PATH="$HOME/.local/bin:$PATH"
eval "$(zoxide init zsh)"
' >> ~/.zshrc

# (待定)智能移动 z
git clone --depth 1 https://github.com/rupa/z.git ${ZSH_CUSTOM:-~/.zsh/plugins}/z
echo '# z
source ~/.zsh/plugins/z/z.sh
' >> ~/.zshrc

# 重启
source ~/.zshrc

# eza
sudo apt update
sudo apt install -y gpg
sudo mkdir -p /etc/apt/keyrings
wget -qO- https://raw.githubusercontent.com/eza-community/eza/main/deb.asc | sudo gpg --dearmor -o /etc/apt/keyrings/gierens.gpg
echo "deb [signed-by=/etc/apt/keyrings/gierens.gpg] http://deb.gierens.de stable main" | sudo tee /etc/apt/sources.list.d/gierens.list
sudo chmod 644 /etc/apt/keyrings/gierens.gpg /etc/apt/sources.list.d/gierens.list
sudo apt update
sudo apt install -y eza
